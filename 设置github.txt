git常用命令： git add . 【提交文件】 
 	     git commit -m '描述' 【生成描述】 
 	     git push origin main/master 【推送服务器】 
	     git branch -a 【查看所有分支】 
 	     git checkout 【分支名称】 
	     git pull --rebase origin main/master  【不同步时，推送git失败后可尝试】
	     git mv oldfiles  newfiles  【修改文件夹名称】
